# netlify.toml

# Pengaturan Build Dasar
# Memberi tahu Netlify cara membangun situs Anda.
[build]
  # Perintah untuk menjalankan build. Sesuai dengan "scripts" di package.json.
  command = "npm run build"
  
  # Direktori yang berisi file hasil build yang akan di-deploy.
  publish = "dist/"
  
  # Direktori yang berisi kode serverless functions Anda.
  # Ini adalah baris KUNCI yang akan memperbaiki error 404 Anda.
  functions = "netlify/functions/"

# Pengaturan spesifik untuk serverless functions
[functions]
  # Menentukan bundler Node.js yang akan digunakan. "esbuild" adalah pilihan
  # modern dan cepat yang direkomendasikan Netlify.
  node_bundler = "esbuild"
